interface User {
  id: number;
  name: string;
  email?: string;   // optional
}

function getById<T>(id: number, items: T[]): T | undefined {
  return items.find((item: any) => item.id === id);
}

const user: User = { id: 1, name: "Alice" };
const found = getById<User>(1, [user]);  // generic preserves type
